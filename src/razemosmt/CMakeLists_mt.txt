#
# CMakeLists_mt.txt — razemOSmt build targets
#
# Usage: add to main CMakeLists.txt:
#   include(src/razemosmt/CMakeLists_mt.txt)
#
# Old version (razemos, hass, launchpad) remains UNCHANGED.
#
# Targets:
#   razemos_kernel   — kernel ($0200–$0E5F), standalone for testing
#   razemos_mt       — shell TASK0 ($0E60+), standalone (requires kernel in RAM)
#   razemos_mt_full  — combined image: kernel + shell, single .rp6502 file
#                      RESET=$0200, loads everything at once
#

cmake_minimum_required(VERSION 3.20)

find_package(Python3 REQUIRED COMPONENTS Interpreter)

# ---------------------------------------------------------------------------
# razemos_kernel — razemOSmt kernel (standalone, for testing)
# ---------------------------------------------------------------------------

add_executable(razemos_kernel)

rp6502_executable(razemos_kernel DATA 0x0200 RESET 0x0260)

target_sources(razemos_kernel PRIVATE
    ${CMAKE_CURRENT_LIST_DIR}/kernel.s
)

target_link_options(razemos_kernel PRIVATE
    -C ${CMAKE_CURRENT_LIST_DIR}/kernel.cfg
)

# ---------------------------------------------------------------------------
# razemos_mt — shell as TASK0 (standalone, requires kernel in RAM)
# ---------------------------------------------------------------------------

add_executable(razemos_mt)

rp6502_executable(razemos_mt DATA 0x0E60 RESET 0x0E60)

target_sources(razemos_mt PRIVATE
    ${CMAKE_CURRENT_LIST_DIR}/shell_mt.c
    ${CMAKE_CURRENT_LIST_DIR}/crt0_shell.s
    ${CMAKE_CURRENT_LIST_DIR}/initmainargs.s
)

target_link_options(razemos_mt PRIVATE
    -C ${CMAKE_CURRENT_LIST_DIR}/shell.cfg
)

# ---------------------------------------------------------------------------
# razemos_mt_full — combined image: kernel ($0200) + shell ($0E60)
#
# Mechanism: rp6502.py create razemos_mt (ELF, no RESET) -- razemos_kernel.rp6502
#   razemos_kernel.rp6502 contains data $0200–$0E5F + RESET=$0260 (kernel_init)
#   razemos_mt (ELF) contains data from $0E60
#   rp6502.py merges both into razemos_mt_full.rp6502 with RESET=$0260
#
# Note: add_binary_file for ELF with data=0x0E60 (no reset/irq) does not set
# $FFFC–$FFFF vectors, so kernel.rp6502 sets RESET=$0260 without conflict.
# ---------------------------------------------------------------------------

set(KERNEL_ROM  ${CMAKE_CURRENT_BINARY_DIR}/razemos_kernel.rp6502)
set(SHELL_ELF   ${CMAKE_CURRENT_BINARY_DIR}/razemos_mt)
set(FULL_ROM    ${CMAKE_CURRENT_BINARY_DIR}/razemos_mt_full.rp6502)

add_custom_command(
    OUTPUT  ${FULL_ROM}
    DEPENDS razemos_kernel razemos_mt
    COMMAND ${Python3_EXECUTABLE}
            ${CMAKE_SOURCE_DIR}/tools/rp6502.py
            -a 0x0E60
            -o ${FULL_ROM}
            create ${SHELL_ELF}
            -- ${KERNEL_ROM}
    COMMENT "Merging kernel + shell into razemos_mt_full.rp6502"
)

add_custom_target(razemos_mt_full ALL
    DEPENDS ${FULL_ROM}
)
