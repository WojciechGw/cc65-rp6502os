#
# CMakeLists_mt.txt — razemOSmt build targets
#
# Usage: add to main CMakeLists.txt:
#   include(src/razemosmt/CMakeLists_mt.txt)
#
# Old version (razemos, hass, launchpad) remains UNCHANGED.
#
# Targets:
#   rmt_kernel       — kernel ($0200–$0F5F), standalone for testing
#   rmt_task0        — shell TASK0 ($1000+), standalone (requires kernel in RAM)
#   rmt_task1        — TASK1 heartbeat ($A000), standalone task binary
#   rmt_full         — combined image: kernel + shell + task1, single .rp6502 file
#                      RESET=$0260, loads everything at once
#

cmake_minimum_required(VERSION 3.20)

find_package(Python3 REQUIRED COMPONENTS Interpreter)

# ---------------------------------------------------------------------------
# razemos_kernel — razemOSmt kernel (standalone, for testing)
# ---------------------------------------------------------------------------

add_executable(rmt_kernel)

rp6502_executable(rmt_kernel DATA 0x0200 RESET 0x0260)

target_sources(rmt_kernel PRIVATE
    ${CMAKE_CURRENT_LIST_DIR}/kernel.s
)

target_link_options(rmt_kernel PRIVATE
    -C ${CMAKE_CURRENT_LIST_DIR}/kernel.cfg
)

# Patch IRQ vector ($FFFE/$FFFF) into rmt_kernel.rp6502 after build.
# rp6502_executable only sets RESET; IRQ address is extracted from linker map.
add_custom_command(TARGET rmt_kernel POST_BUILD
    COMMAND ${Python3_EXECUTABLE} ${CMAKE_CURRENT_LIST_DIR}/patch_irq_vec.py
            ${CMAKE_CURRENT_BINARY_DIR}/rmt_kernel.map
            ${CMAKE_CURRENT_BINARY_DIR}/rmt_kernel.rp6502
    COMMENT "Patching IRQ vector into rmt_kernel.rp6502"
)

# ---------------------------------------------------------------------------
# rmt_task0 — shell as TASK0 (standalone, requires kernel in RAM)
# ---------------------------------------------------------------------------

add_executable(rmt_task0)

rp6502_executable(rmt_task0 DATA 0x1000 RESET 0x1000)

target_sources(rmt_task0 PRIVATE
    ${CMAKE_CURRENT_LIST_DIR}/shell_mt.c
    ${CMAKE_CURRENT_LIST_DIR}/crt0_shell.s
    ${CMAKE_CURRENT_LIST_DIR}/initmainargs.s
    ${CMAKE_CURRENT_LIST_DIR}/kern_calls.s
)

target_link_options(rmt_task0 PRIVATE
    -C ${CMAKE_CURRENT_LIST_DIR}/shell.cfg
)

# ---------------------------------------------------------------------------
# rmt_task1 — TASK1 heartbeat ($A000)
#
# rp6502_executable is NOT used here because it always writes a RESET vector
# ($FFFC-$FFFD) which would conflict when merging into razemos_mt_full.
# Instead the ELF is packaged by a custom command with -a 0xA000 and no -r,
# producing razemos_task1.rp6502 that contains only the task code data.
# ---------------------------------------------------------------------------

add_executable(rmt_task1)

target_sources(rmt_task1 PRIVATE
    ${CMAKE_CURRENT_LIST_DIR}/task1_heartbeat.c
    ${CMAKE_CURRENT_LIST_DIR}/crt0_task.s
    ${CMAKE_CURRENT_LIST_DIR}/kern_calls.s
)

target_link_options(rmt_task1 PRIVATE
    -C ${CMAKE_CURRENT_LIST_DIR}/task1.cfg
)

add_custom_command(TARGET rmt_task1 POST_BUILD
    COMMAND ${Python3_EXECUTABLE}
            ${CMAKE_SOURCE_DIR}/tools/rp6502.py
            -a 0xA000
            -o ${CMAKE_CURRENT_BINARY_DIR}/rmt_task1.rp6502
            create ${CMAKE_CURRENT_BINARY_DIR}/rmt_task1
    COMMENT "Packaging TASK1 (no RESET vector)"
)

# ---------------------------------------------------------------------------
# rmt_task2 — TASK2 counter ($B800)
# ---------------------------------------------------------------------------

add_executable(rmt_task2)

target_sources(rmt_task2 PRIVATE
    ${CMAKE_CURRENT_LIST_DIR}/task2_counter.c
    ${CMAKE_CURRENT_LIST_DIR}/crt0_task.s
    ${CMAKE_CURRENT_LIST_DIR}/kern_calls.s
)

target_link_options(rmt_task2 PRIVATE
    -C ${CMAKE_CURRENT_LIST_DIR}/task2.cfg
)

add_custom_command(TARGET rmt_task2 POST_BUILD
    COMMAND ${Python3_EXECUTABLE}
            ${CMAKE_SOURCE_DIR}/tools/rp6502.py
            -a 0xB800
            -o ${CMAKE_CURRENT_BINARY_DIR}/rmt_task2.rp6502
            create ${CMAKE_CURRENT_BINARY_DIR}/rmt_task2
    COMMENT "Packaging TASK2 (no RESET vector)"
)

# ---------------------------------------------------------------------------
# rmtf — combined image: kernel + shell + task1 + task2
#
#   rmt_kernel.rp6502  — data $0200–$0F5F, RESET=$0260 (kernel_init)
#   rmt_task0 (ELF)    — data from $1000 (TASK0 shell)
#   rmt_task1.rp6502   — data from $A000 (TASK1 heartbeat)
#   rmt_task2.rp6502   — data from $B800 (TASK2 counter)
# ---------------------------------------------------------------------------

set(KERNEL_ROM  ${CMAKE_CURRENT_BINARY_DIR}/rmt_kernel.rp6502)
set(SHELL_ELF   ${CMAKE_CURRENT_BINARY_DIR}/rmt_task0)
set(TASK1_ROM   ${CMAKE_CURRENT_BINARY_DIR}/rmt_task1.rp6502)
set(TASK2_ROM   ${CMAKE_CURRENT_BINARY_DIR}/rmt_task2.rp6502)
set(FULL_ROM    ${CMAKE_CURRENT_BINARY_DIR}/rmtf.rp6502)

add_custom_command(
    OUTPUT  ${FULL_ROM}
    DEPENDS rmt_kernel rmt_task0 rmt_task1 rmt_task2
    COMMAND ${Python3_EXECUTABLE}
            ${CMAKE_SOURCE_DIR}/tools/rp6502.py
            -a 0x1000
            -o ${FULL_ROM}
            create ${SHELL_ELF}
            ${KERNEL_ROM}
            ${TASK1_ROM}
            ${TASK2_ROM}
    COMMENT "Merging kernel + task0(shell) + task1 + task2 into rmtf.rp6502"
)

add_custom_target(rmtf ALL
    DEPENDS ${FULL_ROM}
)
